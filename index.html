// ===== 設定（ここだけ書き換える） ===== const USERNAME = 'staff0721'; // ログインID const PASSWORD = 'yahootire-5963'; // ログインパスワード const GITHUB_PAGES_HOST = 'u-001.github.io'; // GitHub Pagesのホスト名 const GITHUB_PAGES_BASE_PATH = '/yahoo-auction-app'; // リポジトリ名（ユーザー/Organizationページの場合は ''） // ===================================== // Basic認証成功後に発行するセッションCookieの設定。 // JavaScriptのfetch()はCookieを自動送信するため、APIリクエスト時にBasic認証ダイアログが出なくなる。 const SESSION_COOKIE_NAME = 'yahs'; const SESSION_COOKIE_VALUE = btoa(USERNAME + ':' + PASSWORD); // HTTPヘッダーはASCII(ISO-8859-1)のみ許可されるため、日本語等の非ASCII文字は\uXXXXにエスケープする function asciiSafeJSON(obj){ let out = ''; const s = JSON.stringify(obj); for (let i = 0; i < s.length; i++) { const code = s.charCodeAt(i); out += code > 127 ? ('\\u' + code.toString(16).padStart(4, '0')) : s[i]; } return out; } function unauthorized() { return new Response('認証が必要です。', { status: 401, headers: { 'WWW-Authenticate': 'Basic realm="Restricted"' } }); } function json(obj, status = 200) { return new Response(JSON.stringify(obj), { status, headers: { 'Content-Type': 'application/json; charset=utf-8' } }); } // セッションCookieの検証 function hasValidSession(request) { const cookies = request.headers.get('Cookie') || ''; return cookies.split(';').some(c => { const eq = c.indexOf('='); if (eq < 0) return false; return c.slice(0, eq).trim() === SESSION_COOKIE_NAME && c.slice(eq + 1).trim() === SESSION_COOKIE_VALUE; }); } // リフレッシュトークンを使って有効なアクセストークンを取得する。 // DROPBOX_REFRESH_TOKEN + DROPBOX_APP_KEY + DROPBOX_APP_SECRET が設定されている場合はそちらを優先し、 // それらが未設定の場合は従来の DROPBOX_TOKEN（短期トークン）をそのまま使う。 async function getDropboxAccessToken(env) { if (env.DROPBOX_REFRESH_TOKEN && env.DROPBOX_APP_KEY && env.DROPBOX_APP_SECRET) { const res = await fetch('https://api.dropbox.com/oauth2/token', { method: 'POST', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ grant_type: 'refresh_token', refresh_token: env.DROPBOX_REFRESH_TOKEN, client_id: env.DROPBOX_APP_KEY, client_secret: env.DROPBOX_APP_SECRET }) }); const data = await res.json().catch(() => ({})); if (data.access_token) return data.access_token; throw new Error('Dropboxトークン更新失敗: ' + JSON.stringify(data)); } return env.DROPBOX_TOKEN || null; } // ===== 共有データAPI（Cloudflare KVを使った商品リストの共有） ===== // KV Namespaceを「ITEMS_KV」という名前でこのWorkerにバインドしておくこと（Settings → Variables → KV Namespace Bindings） // 注：kv.list()は新しく書き込んだキーが反映されるまで遅延があるため使わない。 // また、共有の「meta」キーを読んでから書き戻す（read-modify-write）方式は、 // 短時間に複数回書き込むとKVの結果整合性の影響で上書き消失することがあるため使わない。 // 代わりに、出品NoはnextNoまでの連番である前提で、item:1～item:(nextNo-1)を直接1件ずつ取得する。 async function readMeta(kv){ const raw = await kv.get('meta'); const meta = raw ? JSON.parse(raw) : {}; if (!meta.usedNos) meta.usedNos = []; if (!meta.nextNo) meta.nextNo = 1; return meta; } async function handleApi(request, env, url) { const kv = env.ITEMS_KV; // GET /api/items : 商品リスト全件を返す if (url.pathname === '/api/items' && request.method === 'GET') { const meta = await readMeta(kv); const items = []; for (let no = 1; no < meta.nextNo; no++) { const val = await kv.get(`item:${no}`); if (val) items.push(JSON.parse(val)); } items.sort((a, b) => (a.order ?? a.no) - (b.order ?? b.no)); return json({ items }); } // PUT /api/items/:no : 1件を保存（追加・更新）。metaへの読み書きは不要。 const itemMatch = url.pathname.match(/^\/api\/items\/(\d+)$/); if (itemMatch && request.method === 'PUT') { const no = itemMatch[1]; const body = await request.text(); await kv.put(`item:${no}`, body); return json({ ok: true }); } // DELETE /api/items/:no : 1件を削除。metaへの読み書きは不要。 if (itemMatch && request.method === 'DELETE') { const no = itemMatch[1]; await kv.delete(`item:${no}`); return json({ ok: true }); } // GET /api/meta : 出品No管理情報（nextNo・usedNos）を返す if (url.pathname === '/api/meta' && request.method === 'GET') { const meta = await readMeta(kv); return json({ nextNo: meta.nextNo, usedNos: meta.usedNos }); } // PUT /api/meta : 出品No管理情報を保存（読み込み不要、そのまま上書き） if (url.pathname === '/api/meta' && request.method === 'PUT') { const incoming = JSON.parse(await request.text()); await kv.put('meta', JSON.stringify({ nextNo: incoming.nextNo, usedNos: incoming.usedNos })); return json({ ok: true }); } // POST /api/dropbox/upload : Dropboxファイルアップロードプロキシ（トークンはWorker Secretで管理） if (url.pathname === '/api/dropbox/upload' && request.method === 'POST') { const token = await getDropboxAccessToken(env); if (!token) return json({ error: 'DROPBOX_TOKEN not configured' }, 500); const path = url.searchParams.get('path') || ''; const body = await request.arrayBuffer(); const res = await fetch('https://content.dropboxapi.com/2/files/upload', { method: 'POST', headers: { 'Authorization': 'Bearer ' + token, 'Dropbox-API-Arg': asciiSafeJSON({ path, mode: 'add', autorename: true, mute: true }), 'Content-Type': 'application/octet-stream' }, body }); if (!res.ok) { const errText = await res.text(); return json({ error: 'Dropboxアップロード失敗: ' + errText }, res.status); } return json(await res.json().catch(() => ({}))); } // POST /api/dropbox/shared-link : Dropbox共有リンク取得・作成プロキシ if (url.pathname === '/api/dropbox/shared-link' && request.method === 'POST') { const token = await getDropboxAccessToken(env); if (!token) return json({ error: 'DROPBOX_TOKEN not configured' }, 500); const { path: folderPath } = JSON.parse(await request.text()); const headers = { 'Authorization': 'Bearer ' + token, 'Content-Type': 'application/json' }; const res = await fetch('https://api.dropboxapi.com/2/sharing/create_shared_link_with_settings', { method: 'POST', headers, body: JSON.stringify({ path: folderPath }) }); if (res.ok) return json(await res.json()); const errData = await res.json().catch(() => ({})); if (errData.error && errData.error['.tag'] === 'shared_link_already_exists') { const res2 = await fetch('https://api.dropboxapi.com/2/sharing/list_shared_links', { method: 'POST', headers, body: JSON.stringify({ path: folderPath, direct_only: true }) }); return json(await res2.json()); } return json(errData, res.status); } return json({ error: 'not found' }, 404); } export default { async fetch(request, env) { const url = new URL(request.url); // APIエンドポイント: Basic認証の代わりにセッションCookieで認証する。 // fetch()はCookieを自動送信するため、ZIP出力中にBasic認証ダイアログが出なくなる。 if (url.pathname.startsWith('/api/')) { if (!hasValidSession(request)) { return json({ error: 'セッションが無効です。ページをリロードしてください。' }, 401); } return handleApi(request, env, url); } // ページリクエスト: Basic認証をチェック const authHeader = request.headers.get('Authorization'); if (!authHeader || !authHeader.startsWith('Basic ')) { return unauthorized(); } let user = '', pass = ''; try { const decoded = atob(authHeader.slice(6)); const idx = decoded.indexOf(':'); user = decoded.slice(0, idx); pass = decoded.slice(idx + 1); } catch (e) { return unauthorized(); } if (user !== USERNAME || pass !== PASSWORD) { return unauthorized(); } // 認証成功: GitHub Pagesにプロキシし、セッションCookieをセット（24時間有効） const targetUrl = `https://${GITHUB_PAGES_HOST}${GITHUB_PAGES_BASE_PATH}${url.pathname}${url.search}`; const resp = await fetch(targetUrl, { method: request.method, headers: request.headers, }); const newHeaders = new Headers(resp.headers); newHeaders.set('Set-Cookie', `${SESSION_COOKIE_NAME}=${SESSION_COOKIE_VALUE}; Path=/; SameSite=Strict; HttpOnly; Max-Age=86400`); return new Response(resp.body, { status: resp.status, headers: newHeaders }); } };
